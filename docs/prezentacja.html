# Bezpośredni wykres z mtcars
ggplot(mtcars, aes(x = hp, y = mpg)) +
  geom_point(size = 4, color = "steelblue", alpha = 0.7) +
  labs(
    title = "Zależność mocy silnika od spalania",
    subtitle = "Im większa moc, tym zazwyczaj większe spalanie",
    x = "Moc (HP - konie mechaniczne)",
    y = "Spalanie (MPG - mile na galon)"
  ) +
  theme_minimal() +
  theme(
    plot.title = element_text(size = 16, face = "bold"),
    axis.title = element_text(size = 12)
  )

Wykres pojemność silnika vs spalanie

library(ggplot2)


ggplot(mtcars, aes(x = disp, y = mpg)) +
  geom_point(size = 4, color = "steelblue", alpha = 0.7) +
  geom_smooth(method = "lm", se = TRUE, color = "darkgreen", fill = "lightgreen") +
  labs(
    title = "Zależność pojemności silnika od spalania",
    subtitle = paste("Korelacja: r =", round(cor(mtcars$disp, mtcars$mpg), 2)),
    x = "Pojemność silnika (cubic inches)",
    y = "Spalanie (miles per gallon)"
  ) +
  theme_minimal() +
  theme(
    plot.title = element_text(size = 16, face = "bold")
  )
